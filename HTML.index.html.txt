    <script>
        let coins = 100, gems = 0;
        let isGaming = false;
        let activeBet = null;

        function toast(m) {
            const t = document.getElementById('toast');
            t.innerText = m; t.style.display = 'block';
            setTimeout(() => t.style.display = 'none', 2000);
        }

        function updateUI() {
            document.getElementById('coin-val').innerText = Math.floor(coins);
            document.getElementById('gem-val').innerText = gems;
            if(coins < 10) { coins += 50; toast("救济金发放：+50"); }
        }

        function switchTab(v, el) {
            if(isGaming) return toast("请先完成当前局！");
            document.querySelectorAll('.game-view').forEach(g => g.classList.remove('active'));
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            document.getElementById('view-'+v).classList.add('active');
            el.classList.add('active');
            activeBet = null;
            document.querySelectorAll('.chip').forEach(c => c.classList.remove('selected'));
        }

        function setBet(b) {
            if(isGaming) return;
            activeBet = b;
            document.querySelectorAll('.chip').forEach((c, i) => {
                c.classList.toggle('selected', b === i || b === c.innerText.split(' ')[0]);
            });
        }

        // --- 老虎机逻辑 ---
        const symbols = ['🍒', '🍋', '🔔', '💎', '7️⃣'];
        function playSlots() {
            if(coins < 10) return;
            isGaming = true; coins -= 10; updateUI();
            
            const results = [];
            const reels = [0,1,2].map(i => document.getElementById('reel-'+i));
            
            reels.forEach((reel, i) => {
                let count = 0;
                const max = 20 + i * 10;
                const timer = setInterval(() => {
                    const sym = symbols[Math.floor(Math.random()*symbols.length)];
                    reel.innerText = sym;
                    count++;
                    if(count >= max) {
                        clearInterval(timer);
                        results[i] = sym;
                        if(results.filter(x => x).length === 3) checkSlots(results);
                    }
                }, 50);
            });
        }

        function checkSlots(res) {
            setTimeout(() => {
                if(res[0] === res[1] && res[1] === res[2]) {
                    coins += 100; toast("🎉 大奖！+100");
                } else if(res[0] === res[1] || res[1] === res[2] || res[0] === res[2]) {
                    coins += 20; toast("中奖！+20");
                } else {
                    toast("未中奖");
                }
                isGaming = false; updateUI();
            }, 300);
        }

        // --- 赛马逻辑 ---
        function playDerby() {
            if(activeBet === null) return toast("请选马！");
            isGaming = true; coins -= 10; updateUI();
            const hs = [0,0,0,0]; 
            const els = [0,1,2,3].map(i => document.getElementById('h-'+i));
            els.forEach(e => e.style.transform = 'translateX(0)');
            const trackW = document.querySelector('.track').offsetWidth - 60;
            const timer = setInterval(() => {
                let win = -1;
                for(let i=0; i<4; i++) {
                    hs[i] += Math.random()*5+1;
                    els[i].style.transform = `translateX(${hs[i]}px)`;
                    if(hs[i] >= trackW && win === -1) win = i;
                }
                if(win !== -1) {
                    clearInterval(timer);
                    setTimeout(() => {
                        if(win === activeBet) { coins += 40; toast("赢了！+40"); }
                        else toast(`失败，冠军是${win+1}号`);
                        isGaming = false; updateUI();
                    }, 500);
                }
            }, 50);
        }

        // --- 骰宝逻辑 ---
        function playSicBo() {
            if(!activeBet) return toast("请下注大小！");
            isGaming = true; coins -= 10; updateUI();
            const faces = ['','⚀','⚁','⚂','⚃','⚄','⚅'];
            const d = [0,0,0].map(() => Math.floor(Math.random()*6)+1);
            d.forEach((v, i) => document.getElementById('d-'+i).innerText = faces[v]);
            const sum = d.reduce((a,b)=>a+b);
            const win = (sum >= 11 && activeBet === 'BIG') || (sum <= 10 && activeBet === 'SMALL');
            setTimeout(() => {
                if(win) { coins += 20; toast(`总和${sum}: 赢了！`); }
                else toast(`总和${sum}: 输了`);
                isGaming = false; updateUI();
            }, 800);
        }

        // --- 8/9点 ---
        function playBac() {
            if(!activeBet) return toast("请下注！");
            isGaming = true; coins -= 10; updateUI();
            const p = (Math.floor(Math.random()*9)+Math.floor(Math.random()*9))%10;
            const d = (Math.floor(Math.random()*9)+Math.floor(Math.random()*9))%10;
            document.getElementById('bac-res').style.display = 'block';
            document.getElementById('p-s').innerText = p;
            document.getElementById('d-s').innerText = d;
            const res = p > d ? 'P' : (d > p ? 'D' : 'T');
            setTimeout(() => {
                if(activeBet === res) { coins += (res==='T'?90:20); toast("赢了！"); }
                else toast("未中奖");
                isGaming = false; updateUI();
            }, 1000);
        }

        // --- 21点 ---
        let bjP = [], bjD = [];
        function playBJ() {
            isGaming = true; coins -= 10; updateUI();
            bjP = [rC(), rC()]; bjD = [rC(), rC()];
            document.getElementById('btn-bj-go').style.display = 'none';
            document.getElementById('bj-ctrl').style.display = 'flex';
            renderBJ(true);
        }
        function rC() { return Math.floor(Math.random()*10)+1; }
        function renderBJ(h) {
            document.getElementById('bj-p').innerText = "玩家: " + bjP.join(' ') + ` (${bjP.reduce((a,b)=>a+b)})`;
            document.getElementById('bj-d').innerText = "庄家: " + (h ? bjD[0]+" ?" : bjD.join(' '));
        }
        function bjHit() {
            bjP.push(rC()); renderBJ(true);
            if(bjP.reduce((a,b)=>a+b) > 21) bjEnd("爆牌了！");
        }
        function bjStand() {
            while(bjD.reduce((a,b)=>a+b) < 17) bjD.push(rC());
            renderBJ(false);
            const ps = bjP.reduce((a,b)=>a+b), ds = bjD.reduce((a,b)=>a+b);
            if(ds > 21 || ps > ds) { coins += 20; bjEnd("你赢了！"); }
            else bjEnd("庄家赢了");
        }
        function bjEnd(m) {
            toast(m); document.getElementById('btn-bj-go').style.display = 'block';
            document.getElementById('bj-ctrl').style.display = 'none';
            isGaming = false; updateUI();
        }

        updateUI();
    </script>
